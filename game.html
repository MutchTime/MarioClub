Игра в стиле Sonic

* { margin: 0; padding: 0; box-sizing: border-box; } body { display: flex; flex-direction: column; align-items: center; font-family: Arial, sans-serif; background-color: #87CEEB; } h1 { margin: 20px; color: #333; } #gameArea { position: relative; width: 800px; height: 400px; background-color: #98FB98; overflow: hidden; border: 2px solid #333; } #sonic { position: absolute; bottom: 10px; left: 50px; width: 40px; height: 40px; background-color: blue; border-radius: 50%; } const sonic = document.getElementById("sonic"); const gameArea = document.getElementById("gameArea"); let isJumping = false; let isMovingRight = false; let isMovingLeft = false; let velocityY = 0; const gravity = 0.5; const speed = 5; function jump() { if (!isJumping) { velocityY = -10; isJumping = true; } } function moveRight() { isMovingRight = true; } function moveLeft() { isMovingLeft = true; } function stopMoveRight() { isMovingRight = false; } function stopMoveLeft() { isMovingLeft = false; } function gameLoop() { let sonicPosition = sonic.getBoundingClientRect(); let gameAreaPosition = gameArea.getBoundingClientRect(); // Движение вправо и влево if (isMovingRight && sonicPosition.right < gameAreaPosition.right) { sonic.style.left = sonicPosition.left + speed + "px"; } if (isMovingLeft && sonicPosition.left > gameAreaPosition.left) { sonic.style.left = sonicPosition.left - speed + "px"; } // Прыжки и гравитация velocityY += gravity; sonic.style.bottom = (parseFloat(sonic.style.bottom) || 0) + velocityY + "px"; // Проверка на приземление if (parseFloat(sonic.style.bottom) <= 10) { sonic.style.bottom = "10px"; velocityY = 0; isJumping = false; } requestAnimationFrame(gameLoop); } // Обработка событий клавиатуры document.addEventListener("keydown", (event) => { if (event.code === "ArrowUp") jump(); if (event.code === "ArrowRight") moveRight(); if (event.code === "ArrowLeft") moveLeft(); }); document.addEventListener("keyup", (event) => { if (event.code === "ArrowRight") stopMoveRight(); if (event.code === "ArrowLeft") stopMoveLeft(); }); gameLoop();
